# 方法一

方法一直接使用 map 比较简单，不再赘述，直接放代码.

**时间复杂度**：`O(N)`，因为所有的数字都要遍历一遍  
**空间复杂度**：`O(N)`，使用 map 存储

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
    const map = new Map();
    for (let i of nums) {
        if (!map.get(i)) {
            map.set(i, 1);
        } else {
            map.set(i, map.get(i) + 1);
        }
    }
    return map.get(target) || 0;
};
```

# 方法二

使用二分查找分别找到左边界和右边界

**时间复杂度**：`O(logn)`  
**空间复杂度**：`O(1)`

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
    let i = 0,
        j = nums.length - 1;
    while (i <= j) {
        let mid = Math.floor((i + j) / 2);
        if (target < nums[mid]) {
            j = mid - 1;
        } else {
            i = mid + 1;
        }
    }

    const right = i;

    if (j >= 0 && nums[j] !== target) return 0;

    i = 0;
    j = right;
    while (i <= j) {
        let mid = Math.floor((i + j) / 2);
        if (target > nums[mid]) {
            i = mid + 1;
        } else {
            j = mid - 1;
        }
    }
    const left = j;

    return right - left - 1;
};
```

# 方法三

方法三是对方法二的改进

```js
/**
 * @param {number[]} nums
 * @param {number} target
 * @return {number}
 */
var search = function (nums, target) {
    return helper(nums, target) - helper(nums, target - 1);
};

// 放回target右边界的下一位
const helper = (nums, target) => {
    let i = 0,
        j = nums.length - 1;

    while (i <= j) {
        let mid = Math.floor((i + j) / 2);
        if (target >= nums[mid]) {
            i = mid + 1;
        } else {
            j = mid - 1;
        }
    }

    return i;
};
```
