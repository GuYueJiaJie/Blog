## 层序遍历

```javascript
/**
 * // Definition for a Node.
 * function Node(val,left,right,next) {
 *    this.val = val;
 *    this.left = left;
 *    this.right = right;
 *    this.next = next;
 * };
 */
/**
 * @param {Node} root
 * @return {Node}
 */
var connect = function(root) {
  if (root === null) return null;
  let node = root;
  let queue = [node];
  while (queue.length > 0) {
    let len = queue.length;
    let pre = null;
    for (let i = 0; i < len; i++) {
      node = queue.shift();
      if (pre !== null) {
        pre.next = node;
      }
      node.left && queue.push(node.left);
      node.right && queue.push(node.right);
      pre = node;
    }
  }
  return root;
};
```
